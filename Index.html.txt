    let currentPoemIndex = 0;
    let currentCategory = 'all';
    let filteredPoems = [...poems];

    function getCategoryName(categoryId) {
        const cat = categories.find(c => c.id === categoryId);
        return cat ? cat.name : categoryId;
    }

    function renderCategories() {
        const categoryList = document.getElementById('categoryList');
        categoryList.innerHTML = categories.map(cat => {
            const count = cat.id === 'all' ? poems.length : poems.filter(p => p.category === cat.id).length;
            return `
                <div class="category-item ${currentCategory === cat.id ? 'active' : ''}" 
                     onclick="filterByCategory('${cat.id}')">
                    <span>${cat.icon} ${cat.name}</span>
                    <span class="category-count">${count}</span>
                </div>
            `;
        }).join('');
    }

    function filterByCategory(categoryId) {
        currentCategory = categoryId;
        
        if (categoryId === 'all') {
            filteredPoems = [...poems];
        } else {
            filteredPoems = poems.filter(p => p.category === categoryId);
        }
        
        renderLibrary();
        renderCategories();
        
        const categoryName = getCategoryName(categoryId);
        document.getElementById('categoryLabel').textContent = `Showing: ${categoryName}`;
        
        toggleCategories();
    }

    function toggleCategories() {
        const dropdown = document.getElementById('categoryDropdown');
        dropdown.classList.toggle('active');
    }

    // Close dropdown when clicking outside
    document.addEventListener('click', function(event) {
        const menu = document.querySelector('.category-menu');
        const dropdown = document.getElementById('categoryDropdown');
        
        if (!menu.contains(event.target) && dropdown.classList.contains('active')) {
            dropdown.classList.remove('active');
        }
    });

    function renderLibrary() {
        const grid = document.getElementById('poemGrid');
        grid.innerHTML = filteredPoems.map((poem, index) => {
            const actualIndex = poems.indexOf(poem);
            const categoryName = getCategoryName(poem.category);
            return `
                <div class="poem-card" onclick="openPoem(${actualIndex})">
                    <h3>${poem.title}</h3>
                    <div class="poem-category-tag">${categoryName}</div>
                    <div class="poem-preview">${poem.preview}</div>
                    <a class="read-more">Read more ‚Üí</a>
                </div>
            `;
        }).join('');
    }

    function openPoem(index) {
        currentPoemIndex = index;
        showReading();
        displayPoem(index);
    }

    function displayPoem(index) {
        const poem = poems[index];
        const categoryName = getCategoryName(poem.category);
        const poemDiv = document.getElementById('currentPoem');
        poemDiv.innerHTML = `
            <img src="${poem.image}" alt="${poem.title}" class="poem-image">
            <h2>${poem.title}</h2>
            <div class="poem-category-tag">${categoryName}</div>
            <div class="poem-content">${poem.content}</div>
        `;
        updateNavButtons();
    }

    function updateNavButtons() {
        document.getElementById('prevButton').disabled = currentPoemIndex === 0;
        document.getElementById('nextButton').disabled = currentPoemIndex === poems.length - 1;
    }

    function navigatePoem(direction) {
        currentPoemIndex += direction;
        if (currentPoemIndex >= 0 && currentPoemIndex < poems.length) {
            displayPoem(currentPoemIndex);
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
    }

    function showLibrary() {
        document.getElementById('libraryView').style.display = 'block';
        document.getElementById('readingView').style.display = 'none';
        document.querySelectorAll('.view-toggle button')[0].classList.add('active');
        document.querySelectorAll('.view-toggle button')[1].classList.remove('active');
    }

    function showReading() {
        document.getElementById('libraryView').style.display = 'none';
        document.getElementById('readingView').style.display = 'block';
        document.querySelectorAll('.view-toggle button')[0].classList.remove('active');
        document.querySelectorAll('.view-toggle button')[1].classList.add('active');
    }

    // Initialize
    renderLibrary();
    renderCategories();
</script>
